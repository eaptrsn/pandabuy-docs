/**
 * @file Checkout.js
 * @description A React component for the checkout screen where users can add extra items , add rewards numbers, and checkout
 */
import React, { useState } from "react";
import SuggestionPanel from "./SuggestionPanel";
import Rewards from "./Rewards";
import TextFormatter from "./TextFormatter";
import { usePoints } from './pointsHelper';

/**
 * The `Checkout` component handles the checkout process, displaying the cart items,
 * total price, and integration with rewards and suggestions systems.
 * 
 * @component
 * @param {Object} props - The component props.
 * @param {string} props.checkoutString - A string representing the current cart items.
 * @param {number} props.totalPrice - The total price of the items in the cart.
 * @param {Function} props.handleDatabaseUpdate - Function to update the database with the cart and total price.
 * @param {Function} props.reset - Function to reset the cart after finalizing the order.
 * @returns {JSX.Element} The rendered Checkout component.
 */
const Checkout = ({ checkoutString, totalPrice, handleDatabaseUpdate, reset }) => {
    const [cost, setCost] = useState(totalPrice)
    const [cart, setCart] = useState(checkoutString)
    const [phone, setPhone] = useState('');

    const { addPoints } = usePoints()

    const handleFinalize = () => {
        handleDatabaseUpdate(cart, cost)
        addPoints(phone, Number(cost.toFixed(0)))
        reset()
    }

    // console.log(typeof cost)
    
    const cartItemStyle = {
        display: "flex",
        alignItems: "center",
        justifyContent: "space-between",
        padding: "0.3em",
        borderBottom: "1px solid #ddd",
        gap: "1em",
    };
    
    const nameStyle = {
        flex: 2, // Allow more space for item name
        textAlign: "left",
        overflow: "hidden",
        textOverflow: "ellipsis",
        whiteSpace: "nowrap",
    };
    
    const priceStyle = {
        display: "flex",
        width: "auto",
        textAlign: "center",
    };
    
    return (
        <div style={{ textAlign: "center", marginTop: "2em", padding: "1em" }}>
        <h1>Checkout:</h1>
        <h2>Total Amount: ${cost.toFixed(2)}</h2>
        <h3>Cart:</h3>
        <TextFormatter text={cart}/>
        <SuggestionPanel setCart={setCart} setCost={setCost}/>
        <Rewards cost={cost} setCost={setCost} phone={phone} setPhone={setPhone}/>
        <button onClick={handleFinalize}>Finalize Order</button>
        </div>
        );
    };

    export default Checkout;
