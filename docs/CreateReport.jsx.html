/**
 * @file CreateReport.js
 * @description A React component that serves as a portal for managers to create various reports (Restock Report, Sales Report, Usage Chart, X Report, Z Report).
 */

import { useState } from "react";
import "./CreateReport.css";

/**
 * CreateReport Component
 * @description Provides a UI for managers to navigate to different types of reports or return to the main manager portal.
 *              Includes logic to restrict generating Z Reports to specific hours unless overridden by the user.
 * @param {Object} props - Component properties.
 * @param {Function} props.setNav - Function to update the current navigation state.
 * @returns {JSX.Element} The rendered CreateReport component.
 */
const CreateReport = ({ setNav }) => {
    /**
     * Handles the click event for the Z Report button.
     * Restricts Z Report generation to the hours between 9 PM and 12 AM, but allows override with user confirmation.
     */
    const handleZReportClick = () => {
        const now = new Date();
        const hours = now.getHours();
        if (hours >= 21 || hours === 0) {
            setNav('z-report');
        } else {
            const proceed = window.confirm(
                "It is not between 9 PM and 12 AM. Generating a Z Report now could mess up the report. Do you want to proceed?"
            );
            if (proceed) {
                setNav('z-report');
            }
        }
    };

    return (
        <div className="containerStyle">
            <h1 className="headerStyle">Manager Portal</h1>
            <div className="buttonContainer">
                <button className="buttonStyle" onClick={() => setNav('restock-report')}>
                    <img src="/src/assets/reports/restock.png" className="buttonImage" alt="Restock Report" />
                    Restock Report
                </button>
                <button className="buttonStyle" onClick={() => setNav('sales-report')}>
                    <img src="/src/assets/reports/sales.png" className="buttonImage" alt="Sales Report" />
                    Sales Report
                </button>
                <button className="buttonStyle" onClick={() => setNav('usage-chart')}>
                    <img src="/src/assets/reports/usage.png" className="buttonImage" alt="Usage Chart" />
                    Usage Chart
                </button>
                <button className="buttonStyle" onClick={() => setNav('x-report')}>
                    <img src="/src/assets/reports/xReport.png" className="buttonImage" alt="X Report" />
                    X Report
                </button>
                <button className="buttonStyle" onClick={handleZReportClick}>
                    <img src="/src/assets/reports/zReport.png" className="buttonImage" alt="Z Report" />
                    Z Report
                </button>
                <button className="logoutButtonStyle" onClick={() => setNav('main')}>
                    <img src="/src/assets/reports/logout.png" className="buttonImage" alt="Back to Manager" />
                    Back to Manager
                </button>
            </div>
        </div>
    );
};

export default CreateReport;
