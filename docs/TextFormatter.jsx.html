/**
 * The `TextFormatter` component takes a string of text and formats it into a list of paragraphs.
 * Each segment of text, separated by commas, is displayed as a paragraph with additional spacing.
 * 
 * @component
 * @param {Object} props - The component props.
 * @param {string} [props.text=""] - The input text to be formatted. Defaults to an empty string.
 * @returns {JSX.Element} The formatted text rendered as a list of paragraphs.
 */
const TextFormatter = ({ text = "" }) => {
    const formattedText = text.split(",").map((segment, index) => (
        <p key={index}>{segment.trim()}<br/><br/></p>
    ));
  
    return <div className="space-y-2">{formattedText}</div>;
  };
  
  export default TextFormatter;
