import { useState, useEffect } from "react";
import { useAuth } from "./AuthProvider";
import KitchenPanel from "./KitchenPanel";

/**
 * Kitchen Component - Displays and manages active cooking orders
 * Automatically refreshes order data every second
 * 
 * @component
 * @returns {JSX.Element} The rendered Kitchen component
 */
const Kitchen = () => {
    const containerStyle = {
        display: 'flex',
        flexWrap: 'wrap',
        padding: '16px',
        backgroundColor: '#f7fafc',
        minHeight: '100vh'
      };
    
    const [orders, setOrders] = useState(null)
    const {user, fetchWithAuth} = useAuth()

    /**
     * Sets up polling for active orders and cleans up on unmount
     * Fetches orders with status 'cooking' every second
     */
    useEffect(() => {
        /**
         * Fetches active orders from the database
         * Only retrieves orders with status 'cooking'
         * 
         * @async
         * @function
         */
        const fetchData = async () => {
            const response = await fetchWithAuth("SELECT * FROM transaction_history WHERE order_status='cooking'")
            // console.log(response)
            setOrders(response)
        }

        fetchData()

        const intervalId = setInterval(fetchData, 1*1000);
        return () => clearInterval(intervalId);
    }, [])

    return ( <div style={containerStyle}>
        {orders 
            ? orders.map(item => {
                // console.log(item)
                return <KitchenPanel key={item.transaction_id} data={item} fetchWithAuth={fetchWithAuth}/>
            })
            : "loading"}
    </div>);
}
 
export default Kitchen;
