import React from 'react';
import "./App.css";
import { useAuth } from './AuthProvider';

/**
 * The `App` component serves as the main entry point for the application.
 * It provides a simple interface with "Login" and "Logout" buttons, utilizing the `useAuth` hook for authentication functionality.
 * 
 * @component
 * @returns {JSX.Element} The rendered App component.
 */
function App() {
    const {user, login, logout} = useAuth()

    return (
        <div style={{
            display: "flex",
            justifyContent: "center",
            alignItems: "center",
            height: "100vh",
            backgroundColor: "white",
            flexDirection: "column"
        }}>
            <div style={{
                display: "flex",
                flexDirection: "column", // Stack buttons vertically
                gap: "20px"               // Adds space between buttons
            }}>
                <button onClick={login} style={buttonStyle}>Login</button>
                <button onClick={logout} style={buttonStyle}>Logout</button>
            </div>
        </div>
    );
}

const buttonStyle = {
    backgroundColor: "black",
    color: "#fff",
    padding: "20px 40px",       // Increased padding for larger button
    fontSize: "1.5rem",         // Larger font size
    border: "none",
    borderRadius: "5px",
    cursor: "pointer",
    width: "200px",             // Optional: Fixed width for consistent sizing
    textAlign: "center"
};

export default App;
