/**
 * @file Keypad.js
 * @description A React component that renders a keypad interface for employee sign-in.
 *              Includes number buttons, a clear button, and an enter button.
 */

import './Keypad.css';

/**
 * Keypad Component
 * @description Renders a keypad for input, including buttons for digits, "clear", and "enter". 
 *              Displays the current input and handles button actions via callback functions.
 * @param {Object} props - Component properties.
 * @param {Function} props.handleKeyPress - Callback function to handle number button presses.
 * @param {Function} props.handleClear - Callback function to clear the input.
 * @param {Function} props.handleSubmit - Callback function to submit the input.
 * @param {string} props.input - The current input value displayed on the keypad.
 * @returns {JSX.Element} The rendered keypad component.
 */
const Keypad = ({ handleKeyPress, handleClear, handleSubmit, input }) => {
    // Array of button labels for the keypad
    const buttons = [
        '1', '2', '3',
        '4', '5', '6',
        '7', '8', '9',
        'clear', '0', 'enter'
    ];

    return (
        <div className="keypad-container">
            <h1 className="keypad-title">Employee Sign-In:</h1>
            <br />
            <div className="keypad-display">
                {input}
            </div>

            <div className="keypad-grid">
                {buttons.map((btn) => {
                    if (btn === 'clear') {
                        return (
                            <button
                                key={btn}
                                className="keypad-button clear-button"
                                onClick={handleClear}
                            >
                                Clear
                            </button>
                        );
                    }
                    if (btn === 'enter') {
                        return (
                            <button
                                key={btn}
                                className="keypad-button enter-button"
                                onClick={handleSubmit}
                            >
                                Enter
                            </button>
                        );
                    }
                    return (
                        <button
                            key={btn}
                            className="keypad-button number-button"
                            onClick={() => handleKeyPress(btn)}
                        >
                            {btn}
                        </button>
                    );
                })}
            </div>
        </div>
    );
};

export default Keypad;
