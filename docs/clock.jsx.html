import React, { useEffect, useState } from 'react';
import { usePoints } from './pointsHelper';
import SuggestionPanel from './SuggestionPanel';

/**
 * The `Clock` component displays the current time, updating every second.
 * It also imports and utilizes point-related methods (`addPoints`, `removePoints`, `checkPoints`) from the `usePoints` hook, although they are not currently used in the component's functionality.
 * 
 * @component
 * @returns {JSX.Element} The rendered clock displaying the current time.
 */
function Clock() {
   const [time, setTime] = useState(new Date());
   const {addPoints, removePoints, checkPoints} = usePoints()
   
   /**
     * Sets up an interval to update the current time every second.
     * Cleans up the interval when the component is unmounted.
     */
   useEffect(() => {
      const timer = setInterval(() => setTime(new Date()), 1000);
      return () => clearInterval(timer); 
   }, []);

   
   return (
      <div>
         <h1>{time.toLocaleTimeString([],{hour12 : false})}</h1>
         {/* <SuggestionPanel/> */}
      </div>
   );
}

export default Clock;
