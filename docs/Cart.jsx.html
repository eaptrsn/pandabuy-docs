/**
 * @file Cart.js
 * @description A React component for displaying a shopping cart, allowing users to view items, remove them, and proceed to checkout.
 */

import React from "react";
import { useNavigate } from "react-router-dom";

/**
 * Cart Component
 * @description Renders a cart view with items, their prices, and a total calculation. 
 *              Users can remove items or proceed to checkout.
 * @param {Object} props - Component properties.
 * @param {Array} props.cart - An array of cart items. Each item contains a name and price.
 * @param {Function} props.handleRemoveFromCart - Function to remove an item from the cart.
 * @param {Function} props.calculateTotal - Function to calculate the total cost of items in the cart.
 * @param {Function} props.buttonStyle - Function to apply dynamic styling to the checkout button.
 * @param {Function} props.setIsCheckingOut - Function to set the checkout state (for navigating to checkout).
 * @returns {JSX.Element} The rendered cart component.
 */
const Cart = ({ cart, handleRemoveFromCart, calculateTotal, buttonStyle, setIsCheckingOut }) => {
  const navigate = useNavigate();

  // Inline styles for the cart and its elements
  const cartStyle = {
    width: "33%",
    backgroundColor: "white",
    color: "#D92323",
    padding: "1em",
    position: "relative",
    top: "10px",
    right: "0px",
    borderTop: "1px solid #ddd",
    borderLeft: "1px solid #ddd",
    borderRadius: "8px",
    boxShadow: "0 4px 8px rgba(0, 0, 0, 0.1)",
  };

  const cartItemStyle = {
    display: "flex",
    alignItems: "center",
    justifyContent: "space-between",
    padding: "0.3em",
    borderBottom: "1px solid #ddd",
    gap: "1em",
  };

  const priceStyle = {
    display: "flex",
    width: "auto",
    textAlign: "center",
  };

  const nameStyle = {
    flex: 2,
    textAlign: "left",
    overflow: "hidden",
    textOverflow: "ellipsis",
    whiteSpace: "nowrap",
  };

  return (
    <div style={cartStyle}>
      <h2>Your Cart ðŸ›’</h2>
      {cart.length > 0 ? (
        cart.map((item, index) => (
          <div key={index} style={cartItemStyle}>
            <span style={nameStyle}>{item.menuname || item.name}</span>
            <span style={priceStyle}>${item.price.toFixed(2)}</span>
            <button
              style={{
                backgroundColor: "transparent",
                border: "none",
                color: "#D92323",
                cursor: "pointer",
              }}
              onClick={() => handleRemoveFromCart(index)}
            >
              Remove
            </button>
          </div>
        ))
      ) : (
        <p>Your cart is empty</p>
      )}
      <h3>Total: ${calculateTotal()}</h3>
      <button
        style={buttonStyle(false)}
        onClick={() => setIsCheckingOut(true)}
      >
        Checkout
      </button>
    </div>
  );
};

export default Cart;
