/**
 * @fileoverview Main entry point for the Panda Express management system
 * Configures routing and authentication providers for the application
 */

import React from 'react'
import ReactDOM from 'react-dom/client'
import { GoogleOAuthProvider } from "@react-oauth/google"
import { BrowserRouter, Routes, Route } from "react-router-dom"

import './index.css'
import { AuthProvider } from './AuthProvider.jsx'
import App from './App.jsx'
import Selection from './Selection.jsx'
import Kiosk from './Kiosk.jsx'
import MenuBoard from './MenuBoard.jsx'
import Employee from './Employee.jsx'
import Kitchen from './Kitchen.jsx'

import ManageEmployees from './ManageEmployees.jsx'
import ManageMenu from './ManageMenu.jsx'
import Clock from './clock.jsx'
import RestockReport from './RestockReport.jsx'

import ZReport from './ZReport.jsx'

import EditIngredients from './editIngredient.jsx'
import NewManagerPage from './newManagerPage.jsx'

/**
 * Application Routes Configuration
 * @typedef {Object} RouteConfig
 * @property {string} path - URL path for the route
 * @property {JSX.Element} element - Component to render at the route
 */

/**
 * Main application structure with the following features:
 * - Google OAuth authentication
 * - React Router for navigation
 * - Custom auth context provider
 * 
 * @component
 */
ReactDOM.createRoot(document.getElementById('root')).render(
  <GoogleOAuthProvider clientId='832110389487-mnq08kcgrvddaii903mufvkop57j912i.apps.googleusercontent.com'>
    <BrowserRouter>
      <AuthProvider>
        <Routes>
          {/* Main Entry */}
          <Route index element={<App/>} />
          <Route path="selection" element={<Selection/>}/>

          {/* Customer Interfaces */}
          <Route path="kiosk" element={<Kiosk/>}/>
          <Route path="menuboard" element={<MenuBoard/>}/>
          <Route path="BowlOrdering" element={<BowlOrdering/>}/>

          {/* Employee Interfaces */}
          <Route path="employee" element={<Employee/>}/>
          <Route path="kitchen" element={<Kitchen/>}/>

          {/* Management Interfaces */}
          <Route path="manager" element={<Manager/>}/>
          <Route path="NewManagerPage" element={<NewManagerPage/>}/>
          <Route path="manage-employees" element={<ManageEmployees/>}/>
          <Route path="manage-menu" element={<ManageMenu/>}/>
          
          {/* Reports and Analytics */}
          <Route path="restock-report" element={<RestockReport/>}/>
          <Route path="sales-report" element={<SalesReport/>}/>
          <Route path="usage-chart" element={<UsageChart/>}/>
          <Route path="x-report" element={<XReport/>}/>
          <Route path="z-report" element={<ZReport/>}/>
          <Route path="create-reports" element={<CreateReport/>}/>
          <Route path="transactionHistory" element={<TransactionHistory/>}/>

          {/* Inventory Management */}
          <Route path="inventory" element={<Inventory/>}/>
          <Route path="edit-ingredients/:menuId" element={<EditIngredients/>}/>

          {/* Utilities */}
          <Route path="current-time" element={<Clock/>}/>
        </Routes>
      </AuthProvider>
    </BrowserRouter>
  </GoogleOAuthProvider>
)
